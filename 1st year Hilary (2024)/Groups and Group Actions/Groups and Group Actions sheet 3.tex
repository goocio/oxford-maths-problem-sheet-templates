\section*{Main course}
\begin{questions}

\question%1
For each part below, provide a group $G$ and a proper, non-trivial subgroup $H$ of $G$ according to the different criteria. Provide a different group $G$ in each case.
\begin{parts}
\part%1a
$G$ is infinite and $H$ is infinite and cyclic;

\part%1b
$G$ is infinite and $H$ is finite and cyclic;

\part%1c
$G$ is infinite and non-Abelian and $H$ is Abelian;

\part%1d
$G$ is infinite and Abelian and $H$ is infinite and not cyclic;

\part%1e
$G$ is infinite and non-Abelian and $H$ is infinite and cyclic.
\end{parts}



\question%2
Let $G$ be a group and $H, K$ subgroups of $G$. Let $H K=\{h k: h \in H, k \in K\}$.
\begin{parts}
\part%2a
Show that $H \cap K$ is a subgroup of $G$.

\part%2b
Give an example where $H \cup K$ is not a subgroup of $G$. Justify your answer.

\part%2c
Give an example where $H K$ is not a subgroup of $G$. Justify your answer.

\part%2d
Show that if $G$ is Abelian then $H K$ is a subgroup of $G$.
\end{parts}



\question%3
Which of the following groups are cyclic? Either find a generator or show that no generator exists. For the cyclic groups, determine how many different generators there are.
\begin{parts}
\part%3a
$\mathbb Z^2$;

\part%3b
$C_2\times C_4$;

\part%3c
$C_3\times C_4$;

\part%3d
$\langle(1\; 2)(3\; 4)(5\; 6),(1\; 4\; 5)(2\; 3\; 6)\rangle\leqslant S_6$;

\part%3e
$\langle(1\; 2\; 3),(4\; 5\; 6)\rangle\leqslant S_6$.
\end{parts}



\question%4
\begin{parts}
\part%4a
By considering partitions, calculate the number of equivalence relations on a set with four elements.

\part%4b
Let $q_{0}=1$ and, for $n \geqslant 1$, let $q_{n}$ denote the number of equivalence relations of a set $X$ with $n$ elements. By considering the possible equivalence classes of the $(n+1)$th element, show that \[
	q_{n+1}=\sum_{k=0}^{n}\binom nk q_{n-k}.
\] Use this recurrence relation to verify your answer for $q_{4}$ from part (a).
\end{parts}



\question%5
Let $G$ be a finite group. We define a relation $\sim$ on $G \setminus\{e\}=\{g \in G: g \neq e\}$ by \[
	g \sim h \text { if and only there exists } k \text { such that } g=h^{k}.
\]
\begin{parts}
\part%5a
Show that $\sim$ is reflexive and transitive.

\part%5b
Show that $\sim$ is symmetric if and only if the order of $g$ is prime for every $g \neq e$.
\end{parts}



\question%6
Let $G$ be a finite group and $H, K$ subgroups of $G$. The map $\phi: H \times K \to H K$ is defined by $(h, k) \to h k$.
\begin{parts}
\part%6a
Show that $\left|\phi^{-1}(g)\right|=|H \cap K|$ for any $g \in H K$. Deduce that \[
	|H K|=\frac{|H||K|}{|H \cap K|}.
\]

\part%6b
Show that if $G$ is Abelian and $H \cap K=\{e\}$ then $H K$ is isomorphic to $H \times K$.
\end{parts}

\end{questions}



\section*{Starter}
\begin{questions}

\question%S1
In the dihedral group $D_{8}$ (with the same notation as in lectures), what is the subgroup generated by $r s$? What is the subgroup $\langle r s, s\rangle$?



\question%S2
In this question we work in the group $\mathbb{Z}$. Find all integers $m$ such that $\langle m\rangle=\langle 3,4,6\rangle$, and find all integers $n$ such that $\langle n\rangle=\langle 3\rangle \cap\langle 4\rangle \cap\langle 6\rangle$.



\question%S3 this one is worth bearing in mind, it comes up in some complicated proofs later
Let $G$ be a group. Define a relation $\sim$ on $G$ via $x \sim y$ if and only if $x=y$ or $x=y^{-1}$. Show that $\sim$ is an equivalence relation. What are the equivalence classes?

\end{questions}

\question%P1
Let $G$ be a group with at least 2 elements. Suppose that $G$ has no proper non-trivial subgroups. Take $x \in G \setminus\{e\}$. Is $x$ a generator for $G$? Is $x^{2}$ a generator for $G$? What can you say about the order of $x$?



\question%P2

\part%P2a

\part%P2b
Is there a non-Abelian group all of whose proper subgroups are Abelian?
\end{parts}
