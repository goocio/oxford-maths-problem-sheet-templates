\question%1
Let $\mathbb{P}(X=k)=1 / n$ for $k=1,2, \ldots, n$. Find the mean and variance of $X$.



\question%2
Suppose that the discrete random variables $X$ and $Y$ have joint probability mass function given by \begin{center}\begin{tabular}{cc||c|c|c}
	 & $X$ & $-1$ & 0 & 1 \\
	$Y$ &  &  &  &  \\
	\hline\hline
	$-1$ & & $\frac{1}{27}$ & $\frac{6}{27}$ & $\frac{2}{27}$  \\
	0 &  & $\frac{2}{27}$ & $\frac{6}{27}$ & $\frac{1}{27}$ \\
	1 &  & $\frac{3}{27}$ & $\frac{2}{27}$ & $\frac{4}{27}$ \\
\end{tabular}\end{center} Find the marginal distributions of $X$ and $Y$. What is the covariance of $X$ and $Y$? Are $X$ and $Y$ independent?



\question%3
Suppose that $X$ and $Y$ are independent Poisson random variables with parameters $\lambda$ and $\mu$ respectively. Find
\begin{parts}
\part%3a
the joint probability mass function $\mathbb{P}(X=k, Y=m)$;

\part%3b
$\mathbb{P}(X+Y=n)$ (what is this distribution?);

\part%3c
$\mathbb{P}(X=k \mid X+Y=n)$ (what is this distribution?);

\part%3d
$\mathbb{E}[X \mid X+Y=n]$.
\end{parts}



\question%4

\begin{parts}
\part%4a
Find $\mathbb{P}(X=k \mid X+Y=n+1)$, for $k \in\{1,2, \ldots, n\}$.

\part%4b
Find the distribution of $\min \{X, Y\}$. [\emph{Hint: consider $\mathbb{P}(\min \{X, Y\}>k)$, and see Question 3(a) of sheet 3.}]
\end{parts}

\question%6
\begin{parts}
\part%6a
Let $X$ and $Y$ be discrete random variables. Show that the following two definitions of independence of $X$ and $Y$ are equivalent: \begin{align*}
	\text{(i) For all }x,y\in\mathbb R,&\quad\mathbb P(X=x,Y=y)=\mathbb P(X=x)\mathbb P(Y=y);\\
	\text{(ii) For all }A,B\subseteq\mathbb R,&\quad\mathbb P(X\in A,Y\in B)=\mathbb P(X\in A)\mathbb{P}(Y\in B).
\end{align*}

\part%6b
Show that if $X$ and $Y$ are independent, then also $f(X)$ and $g(Y)$ are independent for any functions $f, g: \mathbb{R} \to \mathbb{R}$.
\end{parts}



\question%7

\part%7c

\part%7d
$u_{n+1}-3 u_{n}+2 u_{n-1}=1$ with $u_{0}=u_{1}=0$.
\end{parts}
