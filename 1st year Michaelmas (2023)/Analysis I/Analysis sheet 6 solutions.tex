\question%1
For the following choices of $a_{k}$, use the indicated test to establish whether or not $\sum a_{k}$ converges. In each case, comment on possible alternative approaches.
\begin{parts}
\part%1a
$\dfrac{(2 k+1)(3 k-1)}{(k+1)(k+2)^{2}}$\hfill(Comparison Test, limit form)

\part%1b
$\dfrac{1}{k^{\frac{1}{k}}k}$\hfill(Comparison Test, limit form)

\part%1c
$\dfrac{k !}{k^{k}}$\hfill(Ratio Test)

\part%1d
$\displaystyle\binom{2k}k^{-1}(4-10^{-23})^{k}$\hfill(Ratio Test)

\part%1e
$2^{-k} k$ if $k$ is of the form $2^{m}$ and 0 otherwise\hfill(Ratio Test)
\end{parts}



\question%2

\question%4
For each of the following power series $\sum c_{k} x^{k}$, establish which of the following is true: (i) $\sum\left|c_{k} x^{k}\right|$ converges for all $x \in \mathbb{R}$; (ii) $\sum\left|c_{k} x^{k}\right|$ converges only for $x=0$; (iii) $\sum\left|c_{k} x^{k}\right|$ converges for $|x|<R$ and diverges for $|x|>R$ for some $R \in \mathbb{R}^{>0}$, which you should determine. What can you say in each case about the values of $x$ for which $\sum c_{k} x^{k}$ converges?
\begin{parts}
\part%4a
$\displaystyle\sum k^{2020} x^{k}$;

\part%4b
$\displaystyle\sum \frac{x^{k}}{2^{k} k^{4}}$;

\part%4c
$\displaystyle\sum 2^{k} x^{k !}$;

\part%4d
$\displaystyle\sum k^{k} x^{k}$;

\part%4e
$\displaystyle\sum \frac{1}{(4 k) !} x^{2 k}$;

\part%4f
$\displaystyle\sum \sin (k) x^{k}$.
\end{parts}



\question%5
Give either a justification or a counterexample for each of the following statements about real series.
\begin{parts}
\part%5a
If $k a_{k} \to 0$ as $k \to \infty$ then $\sum a_{k}$ converges.

\part%5b
If $\lim _{k \to \infty} \frac{a_{k+1}}{a_{k}}$ exists and equals $L$, where $L>1$, then $\sum a_{k}$ diverges.

\part%5c
If $\sum a_{k}$ converges and $\frac{a_{k}}{b_{k}} \to 1$ then $\sum b_{k}$ converges.
\end{parts}
